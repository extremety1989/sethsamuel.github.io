  "use asm";
  function add(x, y) {
    x = x | 0;

    return x + y;
  }
  return { add: add };
}

WebAssembly AST

WAST is a shared target for compilers

Browsers (and node) can transform WAST into Assembly

(module
  (memory 256 256)
  (export "memory" memory)
  (export "add" $add)
  (func $add (param $0 i32) (param $1 i32) (result i32)
    (i32.add
      (get_local $0)
      (get_local $1)
    )
  )
)
